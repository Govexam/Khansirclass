        enableButtons();
      }

      // Open YouTube link in new tab
      window.open('https://youtube.com/@lbsnaa226?si=1NzPOz1WO2M3UiyG', '_blank');

      document.getElementById('loader').style.display = 'block';

      // Simulate checking process with loader

        document.getElementById('subscribeModal').style.display = 'none';

    }

    // Function to enable all buttons

      document.getElementById('nextPageBtn4').classList.add('enabled');

      }
